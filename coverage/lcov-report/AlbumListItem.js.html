import React, {PropTypes} from 'react';
 
import {getUser} from "../../../utils/user-api";
 
export default AlbumListItem;
 
// const request = require('request');
// let regeneratorRuntime =  require("regenerator-runtime");
 
async function AlbumListItem({album}) {
 
    const user = await getUser(album.userId);
 
    // let list = {};
    // list = user;
    //
    // console.log(list)
 
    return (
      <li className="border-bottom">
        <div className="pull-right">
          <a href="http://jsonplaceholder.typicode.com/photos?albumId=">{album.title}</a>
          <strong> {user.Object}</strong>
        </div>
      </li>
    );
 
 
}
 
AlbumListItem.propTypes = {
    album: PropTypes.shape({
        id: PropTypes.number,
        title: PropTypes.string,
        userId: PropTypes.number,
    }),
};
 
AlbumListItem.defaultProps = {
    album: {},
};
